
pipeline{
    agent any
    stages{
        stage('Intialize'){
            steps{
                bat 'mvn clean install'
            }
        }
        stage('Test'){
            steps{
                bat 'mvn test'
            }
        }
        stage('Build'){
            steps{
                bat 'mvn package'
            }
        }
        stage('Deploy'){
            steps{
                bat 'java -cp target/projectadd1-1.0.SNAPSHOT.jar com.sapient.App'
            }
        }
    }
}
